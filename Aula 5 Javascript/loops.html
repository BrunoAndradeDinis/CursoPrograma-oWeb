Loops em JS

Abra o console apertando F12 e indo na aba console 
            
    // loop-for
    for (let i = 0; i <= 5; i++) {
        console.log(i)
    }

    // Loop-while
    var contador = 0
    while (contador <= 5) {
        console.log(contador)

        contador++
    }
    contador = 0
    // Loop do while
    do {
        console.log(contador)

        contador++
    } while (contador <= 5)
